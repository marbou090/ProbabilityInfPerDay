\section*{本日の問題}

\begin{itemize}
  \item[(1)] $E[\bar{X}]$
  \item[(2)] $V[\bar{X}]$
  \item[(3)] $\bar{X}$の分布
  \item[(4)] $E[\hat{S}^2]$   
\end{itemize}
